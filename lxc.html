Linux Containers (LXC)

Guia Rápido

1- lxc-create -n container -t download

2- vim /var/lib/lxc/container/config
lxc.network.type = veth
lxc.network.link = br0
lxc.network.flags = up
lxc.network.hwaddr = 00:16:3e:7d:ab:06
lxc.rootfs = /var/lib/lxc/container/rootfs
lxc.rootfs.backend = dir
lxc.include = /usr/share/lxc/config/debian.common.conf
lxc.tty = 4
lxc.utsname = container
lxc.arch = amd64

3- chroot /var/lib/lxc/container/rootfs passwd

4- lxc-start -n container

5- lxc-console -n container

6- vim /etc/network/interfaces
auto lo
iface lo inet loopback
auto eth0
iface eth0 inet static
address 192.168.2.45
netmask 255.255.255.0
network 192.168.2.0
broadcast 192.168.2.255
gateway 192.168.2.20

7- vim /etc/resolv.conf
nameserver 8.8.8.8
nameserver 8.8.4.4

8- apt-get install apache2 libapache2-mod-php7.0 php7.0 mysql-server iputils-ping -y

9- dpkg -l | grep dhcp

10- apt-get remove --purge isc-dhcp-client
